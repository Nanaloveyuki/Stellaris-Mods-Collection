namespace = sc

# Finds and assigns the player's first available construction ship.
# This is called on game start and monthly until a valid ship exists.
country_event = {
	id = sc.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		is_ai = no
		NOT = { has_country_flag = sc_companion_assigned }
		NOT = { has_country_flag = sc_companion_greeting_seen }
		any_owned_ship = { is_ship_class = shipclass_constructor }
	}
	immediate = {
		random_owned_ship = {
			limit = { is_ship_class = shipclass_constructor }
			set_ship_flag = sc_sarcastic_companion
			save_event_target_as = sc_companion_ship
		}
		set_country_flag = sc_companion_assigned
		remove_country_flag = sc_waiting_for_companion
		set_country_flag = sc_companion_greeting_seen
		country_event = { id = sc.5 }
		if = {
			limit = { NOT = { has_country_flag = sc_transmission_scheduled } }
			set_country_flag = sc_transmission_scheduled
			country_event = { id = sc.10 days = 730 random = 365 }
		}
	}
}

# Reassigns the companion after the original ship is disbanded, destroyed, or otherwise removed.

	immediate = {

# Monthly safety check. If the companion ship was removed, clear the assignment so a replacement can be found.
country_event = {
	id = sc.3
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		is_ai = no
		has_country_flag = sc_companion_assigned
		NOT = { any_owned_ship = { has_ship_flag = sc_sarcastic_companion } }
	}
	immediate = {
		remove_country_flag = sc_companion_assigned
		remove_country_flag = sc_transmission_scheduled
		if = {
			limit = { any_owned_ship = { is_ship_class = shipclass_constructor } }
			country_event = { id = sc.2 days = 30 }
		}
		else = {
			set_country_flag = sc_waiting_for_companion
		}
	}
}

# Dormant waiting state. Used when the empire has no active companion and no construction ship exists yet.

# Random follow-up scheduler. Most transmissions arrive 2-3 years later, with a rare short follow-up.
# This is called after each visible transmission so there is no predictable calendar date.
country_event = {
	id = sc.20
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		is_ai = no
		has_country_flag = sc_companion_assigned
		NOT = { has_country_flag = sc_transmission_scheduled }
		any_owned_ship = { has_ship_flag = sc_sarcastic_companion }
	}
	immediate = {
		set_country_flag = sc_transmission_scheduled
		random_list = {
			95 = { country_event = { id = sc.10 days = 730 random = 365 } }
			5 = { country_event = { id = sc.10 days = 90 random = 90 } }
		}
	}
}
