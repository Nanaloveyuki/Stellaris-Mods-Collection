# Overrides the vanilla rule of the same name.
# this = excavating country
# from = archaeological-site location (galactic_object)
can_excavate_outside_borders = {
	exists = this
	exists = from
	OR = {
		has_tradition = tr_archivism_frontier_archaeology
		is_nomadic = yes
	}
	OR = {
		# Preserve excavation in unowned systems.
		from = {
			solar_system = { NOT = { exists = owner } }
		}

		# AI Nomads may use research agreements only inside AI-owned borders.
		AND = {
			is_ai = yes
			is_nomadic = yes
			from = { solar_system = { exists = owner } }
			from.solar_system.owner = { is_ai = yes }
			has_research_agreement = from.solar_system.owner
		}

		# Preserve the vanilla options for player-controlled Nomad empires.
		AND = {
			is_ai = no
			is_nomadic = yes
			from = { solar_system = { exists = owner } }
			OR = {
				has_research_agreement = from.solar_system.owner
				has_tradition = tr_archivism_frontier_archaeology
			}
		}
	}
}
