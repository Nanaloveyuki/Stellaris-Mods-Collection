namespace = civspace

# Rare annual check for a valid civilian colonization target.
country_event = {
	id = civspace.400
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		civspace_can_start_colony_project = yes
		is_at_war = no
		OR = {
			AND = {
				has_void_dweller_origin = no
				any_system_within_border = {
					NOT = {
						any_fleet_in_system = {
							exists = owner
							owner = { is_at_war_with = root }
						}
					}
					any_system_planet = {
						is_colonizable = yes
						is_colony = no
						NOT = { has_planet_flag = civspace_colony_reserved }
						exists = space_owner
						space_owner = { is_same_value = root }
						habitability = { who = root.species value >= 0.6 }
					}
				}
			}
			AND = {
				has_void_dweller_origin = yes
				has_technology = tech_habitat_1
				any_system_within_border = {
					NOT = {
						any_fleet_in_system = {
							exists = owner
							owner = { is_at_war_with = root }
						}
					}
					any_system_planet = { civspace_can_host_habitat_anchor = yes }
				}
			}
		}
	}

	immediate = {
		random_list = {
			15 = {
				if = {
					limit = { has_void_dweller_origin = yes }
					country_event = { id = civspace.420 }
				}
				else = { country_event = { id = civspace.421 } }
			}
			85 = {}
		}
	}
}

# Select a normal planetary colony target for non-Void Dweller empires.
country_event = {
	id = civspace.421
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		random_system_within_border = {
			limit = {
				NOT = {
					any_fleet_in_system = {
						exists = owner
						owner = { is_at_war_with = root }
					}
				}
				any_system_planet = {
					is_colonizable = yes
					is_colony = no
					NOT = { has_planet_flag = civspace_colony_reserved }
					exists = space_owner
					space_owner = { is_same_value = root }
					habitability = { who = root.species value >= 0.6 }
				}
			}
			random_system_planet = {
				limit = {
					is_colonizable = yes
					is_colony = no
					NOT = { has_planet_flag = civspace_colony_reserved }
					exists = space_owner
					space_owner = { is_same_value = root }
					habitability = { who = root.species value >= 0.6 }
				}
				save_event_target_as = civspace_colony_target
				set_timed_planet_flag = { flag = civspace_colony_reserved days = 1800 }
			}
		}
		save_event_target_as = civspace_colony_owner
		capital_scope = { save_event_target_as = civspace_colony_origin }
		random_country = {
			limit = { has_country_flag = civspace_sector@root }
			save_event_target_as = civspace_colony_sector
		}
		if = {
			limit = {
				exists = event_target:civspace_colony_target
				exists = event_target:civspace_colony_origin
				exists = event_target:civspace_colony_sector
			}
			if = {
				limit = { has_country_flag = civspace_colonization_approval }
				country_event = { id = civspace.401 }
			}
			else_if = {
				limit = { has_country_flag = civspace_colonization_automatic }
				country_event = { id = civspace.402 }
			}
		}
	}
}

# Dispatch the visible colony convoy.
country_event = {
	id = civspace.402
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = event_target:civspace_colony_target
		exists = event_target:civspace_colony_origin
		exists = event_target:civspace_colony_sector
	}

	immediate = {
		set_timed_country_flag = {
			flag = civspace_colonization_cooldown
			days = 3600
		}
		create_fleet = {
			effect = {
				set_owner = event_target:civspace_colony_sector
				create_ship = {
					name = "civspace_Colony_Convoy_NAME"
					design = "civspace_Colony_Convoy_NAME"
					graphical_culture = root
				}
			}
		}
		last_created_fleet = {
			set_name = "civspace_colony_convoy_fleet"
			set_fleet_flag = civspace_colony_project
			set_fleet_stance = passive
			set_location = {
				target = event_target:civspace_colony_origin
				distance = 12
				angle = random
			}
			queue_actions = {
				wait = 15
				move_to = event_target:civspace_colony_target
				wait = 90
				effect = {
					id = "civspace.402.colonize"
					event_target:civspace_colony_target = {
						planet_event = { id = civspace.403 }
					}
					destroy_fleet = this
				}
			}
			fleet_event = { id = civspace.405 days = 1440 random = 180 }
		}
	}
}

# Revalidate and found the player-owned colony.
planet_event = {
	id = civspace.403
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_colonizable = yes
		is_colony = no
		has_planet_flag = civspace_colony_reserved
		exists = event_target:civspace_colony_owner
		exists = space_owner
		space_owner = { is_same_value = event_target:civspace_colony_owner }
	}

	immediate = {
		create_colony = {
			owner = event_target:civspace_colony_owner
			species = event_target:civspace_colony_owner.species
		}
		set_colony_type = col_city
		create_pop_group = {
			size = 100
			species = event_target:civspace_colony_owner.species
		}
		add_modifier = {
			modifier = civspace_privately_founded_colony
			days = 7200
		}
		set_planet_flag = civspace_civilian_founded_colony
		remove_planet_flag = civspace_colony_reserved
		event_target:civspace_colony_owner = {
			change_variable = { which = civspace_colony_count value = 1 }
			if = {
				limit = { NOT = { has_country_flag = civspace_notifications_none } }
				country_event = { id = civspace.404 }
			}
		}
	}
}

# Revalidate the system, create the vanilla habitat planet, and settle it.
planet_event = {
	id = civspace.424
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_planet_flag = civspace_habitat_reserved
		exists = space_owner
		NOT = { has_anomaly = yes }
		NOR = {
			has_carrier_flag = megastructure
			has_carrier_flag = has_megastructure
		}
		solar_system = {
			NOT = {
				any_system_planet = {
					OR = {
						has_carrier_flag = habitat
						has_carrier_flag = hold_the_line_habitat
					}
				}
			}
			NOT = { has_megastructure = habitat_central_complex_ruined }
			NOT = {
				any_system_megastructure = {
					is_megastructure_type = ring_world_ruined
				}
			}
		}
	}

	immediate = {
		space_owner = { save_event_target_as = civspace_habitat_completion_owner }
		solar_system = {
			if = {
				limit = { root = { is_star = yes } }
				civspace_spawn_habitat_effect = {
					HABITAT_OWNER = event_target:civspace_habitat_completion_owner
					TARGET_PLANET = root
					DISTANCE = 19.798
				}
			}
			else = {
				civspace_spawn_habitat_effect = {
					HABITAT_OWNER = event_target:civspace_habitat_completion_owner
					TARGET_PLANET = root
					DISTANCE = 9.899
				}
			}
		}
		solar_system = { set_star_flag = has_habitat }
		remove_planet_flag = civspace_habitat_reserved
		event_target:civspace_completed_habitat = {
			create_colony = {
				owner = event_target:civspace_habitat_completion_owner
				species = event_target:civspace_habitat_completion_owner.species
			}
			set_colony_type = col_habitat_trade
			create_pop_group = {
				size = 100
				species = event_target:civspace_habitat_completion_owner.species
			}
			add_modifier = {
				modifier = civspace_privately_founded_colony
				days = 7200
			}
			set_planet_flag = civspace_civilian_founded_colony
			set_planet_flag = civspace_civilian_founded_habitat
		}
		event_target:civspace_habitat_completion_owner = {
			change_variable = { which = civspace_colony_count value = 1 }
			if = {
				limit = { NOT = { has_country_flag = civspace_notifications_none } }
				country_event = { id = civspace.425 }
			}
		}
	}
}
